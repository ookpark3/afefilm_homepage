<IfModule mod_rewrite.c>
    RewriteEngine On
    
    # 네이버 소유 확인 HTML 파일에 대한 예외 처리
    RewriteCond %{REQUEST_URI} ^/naver0f6462585491596e56a12ff6a0c48920.html$ [NC]
    RewriteRule ^ - [L]

        # Language-based redirection
    # Check for Korean language
    RewriteCond %{HTTP:Accept-Language} ^ko [NC]
    RewriteRule ^$ /kr/ [L,R=301]

    # Default to English
    RewriteRule ^$ /en/ [L,R=301]

    # www 제거하고, www 없는 URL로 리디렉션
    RewriteCond %{HTTP_HOST} ^www\.(.+)$ [NC]
    RewriteRule ^ https://%1%{REQUEST_URI} [R=301,L]

    # en 또는 kr 경로를 en/ 또는 kr/로 리디렉션
    RewriteRule ^(en|kr)$ /$1/ [R=301,L]

    # 404 페이지는 무한 리디렉션 방지를 위해 예외 처리
    RewriteCond %{REQUEST_URI} !^/404.html$ [NC]

    # 정적 파일 및 404 페이지에 대한 예외 처리
    RewriteCond %{REQUEST_URI} !^/(404|kr|en)(/|$) [NC]
    RewriteCond %{REQUEST_URI} !\.(css|js|jpg|jpeg|png|gif|webp|svg|ico|json|pdf|mp4|mov)$ [NC]

    # /index 요청 시 루트(/)로 리디렉션
    RewriteRule ^index(\.html)?$ / [R=301,L]

    # /kr/index, /en/index 요청을 /kr/ 또는 /en/로 리디렉션
    RewriteCond %{THE_REQUEST} ^[A-Z]{3,}\s/(kr|en)/index(\.html)? [NC]
    RewriteRule ^(kr|en)/index(\.html)?$ /%1/ [R=301,L]

    # en 또는 kr 폴더가 없는 경우에만 /kr/로 리디렉션
    RewriteCond %{REQUEST_URI} !^/(kr|en|404)\b [NC]
    RewriteRule ^([^/]+)\.html$ /kr/$1 [R=301,L]

    # .html 확장자 생략 및 처리
    RewriteCond %{REQUEST_FILENAME}.html -f
    RewriteRule ^(.*/)?([^/]+)$ $1$2.html [L]

    # .html 확장자를 제거한 URL로 리디렉션
    RewriteCond %{THE_REQUEST} ^[A-Z]{3,}\s([^.]+)\.html [NC]
    RewriteRule ^ %1 [R=301,L]

    # HTTP 요청을 HTTPS로 리디렉션
    RewriteCond %{HTTPS} off
    RewriteRule ^ https://%{HTTP_HOST}%{REQUEST_URI} [R=301,L]

</IfModule>


# 404 페이지 설정
ErrorDocument 404 /404.html

    AddOutputFilterByType DEFLATE text/html text/plain text/xml text/css text/javascript application/javascript application/json image/webp
</IfModule>

    Header always set Referrer-Policy "strict-origin-when-cross-origin"
